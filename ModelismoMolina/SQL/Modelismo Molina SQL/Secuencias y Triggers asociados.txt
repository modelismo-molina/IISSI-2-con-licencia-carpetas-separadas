CREATE SEQUENCE sec_mod;

CREATE OR REPLACE TRIGGER crea_id_moderador
BEFORE INSERT ON Moderadores
FOR EACH ROW 
BEGIN

SELECT sec_mod.NEXTVAL INTO:NEW.idModerador FROM DUAL;
END;
